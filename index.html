# Python集成示例

import json

API_KEY = "your_4z_api_key"  # 替换为您的API密钥

headers = {

    "Content-Type": "application/json"

    "model": "gpt-4o",  # 可选gpt-4o, gpt-4, gpt-3.5-turbo等

        {"role": "system", "content": "你是一个专业的AI助手"},
        {"role": "user", "content": "请解释ChatGPT API的主要功能和应用场景"}

    "temperature": 0.7,
    "max_tokens": 1000
}

response = requests.post(API_URL, headers=headers, json=data)
result = response.json()
print(result["choices"][0]["message"]["content"])
        

// Node.js集成示例
const axios = require('axios');

const API_KEY = 'your_4z_api_key'; // 替换为您的API密钥
const API_URL = 'https://api.zzzzapi.com/v1/chat/completions';

async function callChatGPTAPI() {
  try {
    const response = await axios.post(
      API_URL,
      {
        model: 'gpt-4o',
        messages: [
          { role: 'system', content: '你是一个专业的AI助手' },
          { role: 'user', content: '请解释ChatGPT API的主要功能和应用场景' }
        ],
        temperature: 0.7,
        max_tokens: 1000
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        }
      }
    );
    
    console.log(response.data.choices[0].message.content);
  } catch (error) {
    console.error('API调用失败:', error.response ? error.response.data : error.message);
  }
}

callChatGPTAPI();
